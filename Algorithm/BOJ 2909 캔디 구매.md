typedef long long int ll;
using namespace std;
#define INF 1234567890
#define N 200000

const int LIMIT = 1e+9;
ll c, k;

ll powf(ll base, ll exp) {
	ll res = 1;
	while (exp) {
		if (exp & 1)
			res *= base;
		exp >>= 1;
		base *= base;
	}
	return res;
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	cin >> c >> k;

	if (k == 0) {
		cout << c;
		return 0;
	}

	ll mod = c % powf(10, k);
	ll div = c - mod;

	if (mod >= powf(10, k) / 2) {
		cout << div + powf(10, k);
	}
	else {
		cout << div;
	}

	return 0;
}
